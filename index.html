Project — Student Record Management System
Menu-driven console app using structs, arrays, functions, file I/O, and pointers.
Required features
  • Add a new student (name, roll, marks in 3 subjects)
  • Display all students with computed average and letter grade
  • Search student by roll number
  • Find topper — student with highest average
  • Sort students by average (descending) using bubble sort
  • Save all records to binary file on exit
  • Load records from file automatically on startup
  • Menu loop using do-while + switch, exits on choice 6
Complete project code
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define MAX   50
#define FNAME "students.dat"

struct Student {
    int   roll;
    char  name[30];
    float m1, m2, m3;
    float avg;
    char  grade;
};

struct Student db[MAX];
int           count = 0;

char calcGrade(float avg) {
    if(avg >= 90) return 'O';
    if(avg >= 80) return 'A';
    if(avg >= 70) return 'B';
    if(avg >= 60) return 'C';
    if(avg >= 40) return 'D';
    return 'F';
}

void addStudent() {
    if(count >= MAX) { printf("Database full!\n"); return; }
    struct Student *s = &db[count];
    printf("Roll No  : "); scanf("%d", &s->roll);
    printf("Name     : "); scanf("%s", s->name);
    printf("3 Marks  : "); scanf("%f %f %f", &s->m1, &s->m2, &s->m3);
    s->avg   = (s->m1 + s->m2 + s->m3) / 3.0f;
    s->grade = calcGrade(s->avg);
    count++;
    printf("Added! Avg: %.2f  Grade: %c\n", s->avg, s->grade);
}

void displayAll() {
    if(count == 0) { printf("No records.\n"); return; }
    printf("\n%-5s %-20s %6s %6s %6s %7s %5s\n",
           "Roll","Name","Sub1","Sub2","Sub3","Avg","Grade");
    printf("---------------------------------------------------\n");
    int i;
    for(i=0; i<count; i++)
        printf("%-5d %-20s %6.1f %6.1f %6.1f %7.2f %5c\n",
               db[i].roll, db[i].name,
               db[i].m1, db[i].m2, db[i].m3,
               db[i].avg, db[i].grade);
}

void searchByRoll() {
    int r, i;
    printf("Enter Roll: "); scanf("%d", &r);
    for(i=0; i<count; i++)
        if(db[i].roll == r) {
            printf("Found: %s | Avg: %.2f | Grade: %c\n",
                   db[i].name, db[i].avg, db[i].grade);
            return;
        }
    printf("Roll %d not found.\n", r);
}

void findTopper() {
    if(count==0) { printf("No records.\n"); return; }
    int i, top=0;
    for(i=1; i<count; i++)
        if(db[i].avg > db[top].avg) top=i;
    printf("Topper: %s | Roll: %d | Avg: %.2f | Grade: %c\n",
           db[top].name, db[top].roll, db[top].avg, db[top].grade);
}

void sortByAvg() {
    int i, j; struct Student tmp;
    for(i=0; i<count-1; i++)
        for(j=0; j<count-i-1; j++)
            if(db[j].avg < db[j+1].avg) {
                tmp=db[j]; db[j]=db[j+1]; db[j+1]=tmp;
            }
    printf("Sorted (highest first):\n");
    displayAll();
}

void saveToFile() {
    FILE *fp = fopen(FNAME, "wb");
    if(!fp) { printf("File error!\n"); return; }
    fwrite(&count, sizeof(int), 1, fp);
    fwrite(db, sizeof(struct Student), count, fp);
    fclose(fp);
    printf("%d record(s) saved.\n", count);
}

void loadFromFile() {
    FILE *fp = fopen(FNAME, "rb");
    if(!fp) { printf("No saved data.\n"); return; }
    fread(&count, sizeof(int), 1, fp);
    fread(db, sizeof(struct Student), count, fp);
    fclose(fp);
    printf("%d record(s) loaded.\n", count);
}

int main() {
    int ch;
    loadFromFile();
    do {
        printf("\n=== Student Records ===\n");
        printf("1.Add  2.Display  3.Search\n");
        printf("4.Topper  5.Sort  6.Save&Exit\n");
        printf("Choice: "); scanf("%d", &ch);
        switch(ch) {
            case 1: addStudent();   break;
            case 2: displayAll();   break;
            case 3: searchByRoll(); break;
            case 4: findTopper();   break;
            case 5: sortByAvg();    break;
            case 6: saveToFile();   break;
            default: printf("Invalid!\n");
        }
    } while(ch != 6);
    return 0;
}
