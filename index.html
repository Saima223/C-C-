Project — C++ Student Management System

Menu-driven app: Classes, Inheritance, Virtual functions, Operator Overloading, File I/O, Exceptions.

Structure
#include <iostream>
#include <fstream>
#include <string>
#include <vector>
using namespace std;

// Ch 14 — Base class with pure virtual
class Person {
protected:
    string name; int age;
public:
    Person(string n, int a) : name(n), age(a) {}
    virtual void display() = 0;  // pure virtual
    virtual ~Person() {}
};

// Ch 18 — Derived class
class Student : public Person {
    int   roll;
    float m[3];
public:
    Student(string n, int a, int r, float m1, float m2, float m3)
        : Person(n, a), roll(r) { m[0]=m1; m[1]=m2; m[2]=m3; }
    float avg()   { return (m[0]+m[1]+m[2])/3; }
    char  grade() { return avg()>=80?'A':avg()>=60?'B':avg()>=40?'C':'F'; }
    void  display() override;
    // Ch 17 — Operator overloading
    bool operator>(Student& s) { return avg() > s.avg(); }
    friend ostream& operator<<(ostream& o, Student& s);
    friend istream& operator>>(istream& i, Student& s);
};

int main() {
    vector<Student*> students;
    // Ch 20 — exception handling
    try {
        ifstream f("records.txt");
        if(!f) throw runtime_error("File not found");
    } catch(exception& e) { cout << e.what() << endl; }
    // Menu with do-while + switch
    int ch;
    do {
        cout << "\n1.Add 2.Display 3.Search 4.Topper 5.Save 0.Exit\n";
        cin >> ch;
        switch(ch) { /* implement each case */ }
    } while(ch != 0);
}
Grading checklist
  • Compiles with zero errors and zero warnings
  • Person base class with pure virtual display() — polymorphism works via base pointer
  • Student derived correctly — constructor chain with Person(n,a) call
  • Average and grade correct (A≥80, B≥60, C≥40, F below)
  • operator> compares by average; operator<< prints cleanly with cout
  • File saves on exit, loads on startup — data survives close and reopen
  • try-catch handles missing file and invalid input gracefully
  • Menu loops until user enters 0
  • Every class and function has a comment explaining its purpose
